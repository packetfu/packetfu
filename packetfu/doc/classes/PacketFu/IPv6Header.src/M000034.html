                def ipv6_saddr=(str)
                        arr = str.split(':').collect {|x| x.to_i(16)}
                        self.ipv6_src.a1 = arr[0]
                        self.ipv6_src.a2 = arr[1]
                        self.ipv6_src.a3 = arr[2]
                        self.ipv6_src.a4 = arr[3]
                end
